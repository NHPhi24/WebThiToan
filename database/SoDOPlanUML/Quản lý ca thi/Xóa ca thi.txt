@startuml
autonumber
skinparam style strictuml
skinparam sequenceMessageAlign center

actor "Giáo viên" as Actor
boundary "Giao diện (UI)" as UI
control "Hệ thống" as System
database "DB" as DB

title Sơ đồ tuần tự: Xóa ca thi

== Truy cập danh sách ==
Actor -> UI: Chọn "Quản lý ca thi"
UI -> System: Yêu cầu lấy danh sách ca thi
System -> DB: Query danh sách ca thi
DB --> System: Trả kết quả dữ liệu
System --> UI: Hiển thị danh sách ca thi

== Kiểm tra trạng thái & Thực hiện xóa ==
System -> System: Kiểm tra trạng thái ca thi

alt Trạng thái khác "Đã kết thúc"
    System --> UI: Chỉ hiển thị nút "Xem chi tiết"
    UI --> Actor: Hiển thị giao diện (Không có nút Xóa)
else Trạng thái là "Đã kết thúc"
    System --> UI: Hiển thị thêm nút "Xóa"
    UI --> Actor: Hiển thị giao diện có nút "Xóa"
    
    Actor -> UI: Chọn "Xóa"
    UI -> UI: Hiển thị Popup xác nhận xóa
    
    alt đóng (Hủy xóa)
        Actor -> UI: Chọn "Không"
        UI -> UI: Đóng Popup
    else có (Xác nhận xóa)
        Actor -> UI: Chọn "Có"
        UI -> System: Gửi yêu cầu xóa ca thi (ID)
        System -> DB: Thực hiện xóa khỏi CSDL (DELETE)
        DB --> System: Xác nhận thành công
        System --> UI: Thông báo xóa thành công
        UI --> Actor: Hiển thị thông báo & cập nhật danh sách
    end
end
@enduml
